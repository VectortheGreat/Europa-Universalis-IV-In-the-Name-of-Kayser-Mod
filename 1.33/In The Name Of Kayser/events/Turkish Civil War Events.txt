country_event = {		#Ankara Rebels
	id = tcwevents.1
	title = tcwevents.1_title
	desc = tcwevents.1_desc
	picture = REVOLUTION_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		had_country_flag = {
			flag = civilwar_start
			days = 30
		}
		NOT = { ai = yes }
	}

	option = {
		name = tcwevents.1a
		set_country_flag = rebels_1
		clr_country_flag = civilwar_start
		if = {
			limit = {
				any_owned_province = {
					area = ankara_area
				}
				NOT = { ai = yes }
			}
			random_owned_province = {
				limit = {
					area = ankara_area
				}
				spawn_rebels = {
					type = sunni_rebels
					size = 10
				}
			}
		}
		ankara_area = {
			add_province_modifier = {
					name = ink_000
					duration = 7300
			}
		}
	}
}
