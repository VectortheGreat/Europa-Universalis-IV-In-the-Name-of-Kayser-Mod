		provinces_to_highlight = {

		trigger = {
			OR = {
				AND = {
					calc_true_if = {

							has_opinion = {

							}

						amount = 3
					}
					calc_true_if = {
						all_elector = {

					}

		}

			add_country_modifier = {

		icon = inkhre

		position = 9
		trigger = {
			OR = {
				OR = {
					hre_religion = protestant

			}
		}
		effect = {
			if = {
				limit = {

					limit = {

						NOT = { total_development = 200 }

						id = itest.5

				}
			}
			else_if = {
				limit = {
					is_emperor = yes

					limit = {
						is_part_of_hre = yes
						NOT = { religion = ROOT }
						NOT = { total_development = 200 }
						NOT = { tag = ITR }
						NOT = { is_emperor = yes }

						id = itest.555

				}
			}
